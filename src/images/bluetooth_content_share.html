const passwordRegex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{8,}$/;

const isValidPassword = password => passwordRegex.test(password);

console.log(isValidPassword("Password1!")); // true
console.log(isValidPassword("weak")); // false
console.log(isValidPassword("STRONGPASSWORD")); // false
console.log(isValidPassword("12345678")); // false
